### 2. CLI smoke tests

All item-referencing commands use UUIDs or unique hex prefixes.
The script captures IDs from `add` output for subsequent commands.

```sh
DB=$(mktemp /tmp/smoke-XXXXXX.ag)
A=agenda-cli

# Basic item lifecycle
ID1=$(cargo run --bin $A -- --db "$DB" add "Buy groceries" 2>/dev/null | awk '{print $2}')
ID2=$(cargo run --bin $A -- --db "$DB" add "Fix bug 42" 2>/dev/null | awk '{print $2}')
cargo run --bin $A -- --db "$DB" list              # should default to "All Items" view (PR #53)

# Category + assignment
cargo run --bin $A -- --db "$DB" category create "Status"
cargo run --bin $A -- --db "$DB" category create "In Progress" --parent Status
cargo run --bin $A -- --db "$DB" category assign "$ID2" "In Progress"
cargo run --bin $A -- --db "$DB" show "$ID2"       # verify In Progress assignment

# Item links and cycle prevention (PR #54)
cargo run --bin $A -- --db "$DB" link depends-on "$ID1" "$ID2"
cargo run --bin $A -- --db "$DB" link depends-on "$ID2" "$ID1" 2>&1 | grep -i cycle
# ^ should error with cycle detection

# Numeric categories (PR #57)
cargo run --bin $A -- --db "$DB" category create "Cost" --type numeric
cargo run --bin $A -- --db "$DB" category set-value "$ID1" "Cost" 25.50

# View create + show columns (PR #56)
cargo run --bin $A -- --db "$DB" view create "StatusView" --include "In Progress"
cargo run --bin $A -- --db "$DB" view show "StatusView"
# ^ should display columns line if the view has columns configured

# View clone (PR #55 — also available via CLI)
cargo run --bin $A -- --db "$DB" view clone "All Items" "My Items"
cargo run --bin $A -- --db "$DB" view show "My Items"

# Note: view add-column and summary-fn configuration are TUI-only.
# Column/summary features (PRs #56, #57) are tested via unit tests
# and TUI smoke tests below.

rm -f "$DB"
```
